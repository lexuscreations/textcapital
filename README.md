function SubmitQuiz()

{

    var currentQuest = parseInt($('#cur_ques').val());



    //code for get user ans and correct ans

    var userAnsID = [];

    userAnsID[0] = $('input[name=Ans'+currentQuest+']:checked').val();    

    

      if(userAnsID != "" && userAnsID.length > 0)

      {

        if($("#UserAnsFlag_"+currentQuest).val()=="0")

        {

            $("#UserAnsFlag_"+currentQuest).val("1");



            $TotalAns = $("#totalAnswered").val();

            $TotalAns++;



            $("#totalAnswered").val($TotalAns);

            $("#AnsSoFar").html($TotalAns);

        }    

      }

      else

      {

        if($("#UserAnsFlag_"+currentQuest).val()=="1")

        {

            $("#UserAnsFlag_"+currentQuest).val("0");



            $TotalAns = $("#totalAnswered").val();

            $TotalAns--;



            $("#totalAnswered").val($TotalAns);

            $("#AnsSoFar").html($TotalAns);

        }    

      }



      //save review question later

      var QuestionReviewLater = 0;

      if($('#reviewLater_'+currentQuest).is(':checked'))

      {

        var QuestionReviewLater = 1;

      }



      var unansweredQues = 1 - $("#AnsSoFar").html();



      bootbox.confirm("<h4 class='popup-head-title'>Are you sure you wish to end your exam?</h4><p class='popup-total-answer'>Total Answered : "+$("#AnsSoFar").html()+"</p><p class='popup-total-unanswer'>Total Unanswered : "+unansweredQues, function(quizSubmitConfirm) {    

        if (quizSubmitConfirm) {



          var QuestionID = $('#questionID'+currentQuest).val();

          var QuizAnswerMasterID = $('#QuizAnswerMasterID'+currentQuest).val();



          $.ajax({

            type: "POST",

            url: BASEURL+"quiz/saveQuizQuestion",

            data: {'QuestionID':QuestionID,'userAnsID': userAnsID,'QuestionReviewLater':QuestionReviewLater,'QuizAnswerMasterID':QuizAnswerMasterID,'RemainingTime':count,'ActivateTimer':ActivateTimer},

            cache: false,    

            success: function(InsertedData) {

              if(InsertedData != "" && InsertedData > 0)

              {

                //alert(rightClick+'----'+copy+'---'+cut+'---'+past+'---'+print_screen+'---'+f12);

                $.ajax({

                type: "POST",

                url: BASEURL+"quiz/SubmitQuiz",

                data: {'rightClick':rightClick,'copy': copy,'cut':cut,'past':past,'print_screen':print_screen,'f12':f12},

                cache: false,    

                success: function(SuccessData) {

                  window.location = BASEURL + "quiz/quiz_submit";

                }

                });

              }

              else

              {

                bootbox.alert("Ooops, Something went Wrong! Your quiz was terminated. Please resume the quiz to continue.", function() 

                {

                  window.location = BASEURL + "quiz_track";

                });

              }

              return false;

            }

          });

        }

    });  

}
